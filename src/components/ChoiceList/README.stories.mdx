# Choice list

A choice list lets you create a list of grouped radio buttons or checkboxes.
Use this component if you need to group together a related list of interactive choices.

<br/><br/>

### Single choice list

Allows merchants to select one option from a list.

* Make sure all options are an either/or choice.

export const singleChoiceList = (args, { argTypes }) => ({
  components: { ChoiceList },
  template: `
  <div>
    <ChoiceList
      name="singleChoiceList"
      v-model="value"
      :choices="choices"
    >
      Company name
    </ChoiceList>
    <p style="margin-top: 30px"> Value: {{ value }} </p>
  </div>
  `,
  data() {
    return {
      value: 'hidden',
      choices: [
        {label: 'Hidden', value: 'hidden'},
        {label: 'Optional', value: 'optional'},
        {label: 'Required', value: 'required'},
      ]
    };
  },
});
singleChoiceList.parameters = {
  docs: {
    source: false,
    transformSource: (source) => {
      const tmpSource = source
        .replace(/value="hidden"/, 'v-model="value"')
        .replace(/\'\[.*\]\'/, '"choices"')
        .replace(/Value: hidden/, 'Value: {{ value }}');
      const fullSource = dedent`
        ${tmpSource}\n
        data() {
          return {
            value: 'hidden',
            choices: [
              {label: 'Hidden', value: 'hidden'},
              {label: 'Optional', value: 'optional'},
              {label: 'Required', value: 'required'},
            ]
          };
        },
      `;
      return fullSource;
    }
  },
}

### Multi-choice list

Allows merchants to select multiple options from a list.

* Avoid options that are an either/or choice.

export const multiChoiceList = (args, { argTypes }) => ({
  components: { ChoiceList },
  template: `
  <div>
    <ChoiceList
      name="multi"
      v-model="value"
      :allowMultiple="true"
      :choices="choices"
    >
      Company name
    </ChoiceList>
    <p style="margin-top: 30px"> Value: {{ value }} </p>
  </div>
  `,
  data() {
    return {
      value: ['shipping'],
      choices: [
        {
          label: 'Use the shipping address as the billing address by default',
          helpText: 'Reduces the number of fields required to check out. The billing address can still be edited.',
          value: 'shipping',
        },
        {
          label: 'Require a confirmation step',
          helpText: 'Customers must review their order details before purchasing.',
          value: 'confirmation',
        },
      ]
    };
  },
});
multiChoiceList.parameters = {
  docs: {
    source: false,
    transformSource: (source) => {
      const tmpSource = source
        .replace(/:value=\'\[\"shipping\"\]\'/, 'v-model="value"')
        .replace(/\'\[.*\]\'/, '"choices"')
        .replace(/<p>.*<\/p>/s, '<p> Value: {{ value }} </p>')
      const fullSource = dedent`
        ${tmpSource}\n
        data() {
          return {
            value: ['hidden'],
            choices: [
              {label: 'Hidden', value: 'hidden'},
              {label: 'Optional', value: 'optional'},
              {label: 'Required', value: 'required'},
            ]
          };
        },
      `;
      return fullSource; 
    }
  },
}

### Single-choice or multi-choice list with children content (rendered only when selected)

Use when you need merchants to view and/or interact with additional content under a choice.
The content is only rendered when the choice is selected. Works for both single-choice and multi-choice list.

export const renderChildWithConditional = (args, { argTypes }) => ({
  components: { ChoiceList },
  template: `
  <div>
    <ChoiceList
      name="renderChildWithConditional"
      v-model="selected"
      :choices="choices"
      @change="handleChange"
      onChange=""
    >
      Company name
    </ChoiceList>
  </div>
  `,
  data() {
    return {
      selected: 'none',
      choices: [
        {label: 'None', value: 'none'},
        {label: 'Minimum purchase', value: 'minimum_purchase'},
        {
          label: 'Minimum quantity',
          value: 'minimum_quantity',
          renderChildren: '',
        },
      ],
    };
  },    
  methods: {
    handleChange() {
      const renderFieldPosition = this.choices
        .findIndex(obj => obj.value === 'minimum_quantity');
      this.choices[renderFieldPosition].renderChildren = this.selected === 'minimum_quantity'
        ? '<p>This is child content of Minimum quantity option</p>'
        : '';
    }
  },
});
renderChildWithConditional.parameters = {
  docs: {
    source: false,
    transformSource: (source) => {
      const tmpSource = source
        .replace(/value="none"/, 'v-model="selected"')
        .replace(/\'\[.*\]\'/, '"choices"')
        .replace(/onChange=""/, '@change="handleChange"')
      const fullSource = dedent`
        ${tmpSource}\n
        data() {
          return {
            selected: 'none',
            choices: [
              {label: 'None', value: 'none'},
              {label: 'Minimum purchase', value: 'minimum_purchase'},
              {
                label: 'Minimum quantity',
                value: 'minimum_quantity',
                renderChildren: '',
              },
            ],
          };
        },
        methods: {
          handleChange() {
            const renderFieldPosition = this.choices
              .findIndex(obj => obj.value === 'minimum_quantity');
            this.choices[renderFieldPosition].renderChildren = this.selected === 'minimum_quantity'
              ? '<p>This is child content of Minimum quantity option</p>'
              : '';
          },
        },
      `;
      return fullSource;
    },
  },
}
