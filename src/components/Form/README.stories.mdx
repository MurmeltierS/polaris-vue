export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { Form, FormLayout, Checkbox, Button, TextField },
  data() {
    return {
      checkbox: false,
      textField: '',
    };
  },
  methods: {
    handleSubmit(event) {
      event.preventDefault();
      this.checkbox = false;
      this.textField = '';
      alert('Form submitted');
    },
  },
  template: `<Form @submit="handleSubmit">
    <FormLayout>
      <Checkbox slot="field-newsletter" id="signUpCheckbox" v-model="checkbox">
        <template slot="label">Sign up for the Polaris newsletter</template>
      </Checkbox>
      <TextField slot="field-email" v-model="textField" type="email" autoComplete="email">
        <template slot="label">Email</template>
        <template slot="help-text">
          <span>
            We'll use this email address to inform you on future changes to
            Polaris.
          </span>
        </template>
      </TextField>
      <Button slot="field-submitBtn" submit>Submit</Button>
    </FormLayout>
  </Form>`,
});
